Q5: Armstrong Number
import java.util.*;

public class Q5_ArmstrongNumber {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), temp = n, sum = 0;
        int digits = Integer.toString(n).length();

        while (temp > 0) {
            int r = temp % 10;
            sum += Math.pow(r, digits);
            temp /= 10;
        }
        if (sum == n) System.out.println("Armstrong Number");
        else System.out.println("Not Armstrong Number");
    }
}

Q12: Strong Number
import java.util.*;

public class Q12_StrongNumber {
    static int fact(int n) {
        int f = 1;
        for (int i = 2; i <= n; i++) f *= i;
        return f;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), temp = n, sum = 0;

        while (temp > 0) {
            int r = temp % 10;
            sum += fact(r);
            temp /= 10;
        }

        if (sum == n) System.out.println("Strong Number");
        else System.out.println("Not Strong Number");
    }
}

Q13: Magic Number
import java.util.*;

public class Q13_MagicNumber {
    static int sumDigits(int n) {
        int sum = 0;
        while (n > 0) {
            sum += n % 10;
            n /= 10;
        }
        return sum;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        while (n > 9) n = sumDigits(n);

        if (n == 1) System.out.println("Magic Number");
        else System.out.println("Not Magic Number");
    }
}

Q16: GCD (Euclidean Algorithm)
import java.util.*;

public class Q16_GCD {
    static int gcd(int a, int b) {
        while (b != 0) {
            int t = b;
            b = a % b;
            a = t;
        }
        return a;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt(), b = sc.nextInt();
        System.out.println(gcd(a, b));
    }
}

Q17: Swap First and Last Digit
import java.util.*;

public class Q17_SwapFirstLast {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.next();
        if (s.length() == 1) {
            System.out.println(s);
            return;
        }

        char[] arr = s.toCharArray();
        char temp = arr[0];
        arr[0] = arr[arr.length - 1];
        arr[arr.length - 1] = temp;

        System.out.println(new String(arr));
    }
}
